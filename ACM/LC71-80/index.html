    LC72¶

    https://leetcode.cn/problems/edit-distance/

    /*动态规划：
 dp数组：dp[m+1][n+1];
    dp[i][j] 表示 字符串word1[0,i)与word2[0,j)的编辑距离；
 传递公式：
    if(word1[i-1]==word2[i-1]){// 字符相同，此时word1[i]与word2[j]分别加入不影响编辑距离
        dp[i][j] = dp[i-1][j-1];
    }
    else{// 字符不同，此时比较并取以下操作最小值：① word1增加1字符、② word2增加1字符、③ 替换1字符；
        dp[i][j] = _min(dp[i-1][j]+1, dp[i][j-1]+1, dp[i-1][j-1]+1);
    }

    综上：
    int sign = word1[i-1]==word2[i-1] ? 0 : 1;
    dp[i][j] = _min(dp[i-1][j]+1, dp[i][j-1]+1, dp[i-1][j-1]+sign);

*/
class Solution {
public:
    // 求三数中最小数
    int _min(int a, int b, int c){
        a = a<b ? a : b;
        return a<c ? a : c;
    }

    int minDistance(string word1, string word2) {
        int m = word1.size(), n = word2.size();
        // 有一个字符串为空串
        if (n * m == 0) return n + m;
        // dp 数组
        vector<vector<int>> dp(m+1, vector<int>(n+1));
        // 边界状态初始化
        for(int i=0; i<=m; ++i) dp[i][0] = i;
        for(int j=0; j<=n; ++j) dp[0][j] = j;
        int sign = 0;
        for(int i=1; i<=m; ++i){
            for(int j=1; j<=n; ++j){
                sign = word1[i-1]==word2[j-1] ? 0 : 1;
                dp[i][j] = _min(dp[i-1][j]+1, dp[i][j-1]+1, dp[i-1][j-1]+sign);
            }
        }
        return dp[m][n];
    }
};
